 41:      * dynamic
 42:      *
 43:      * @param Hook\Model\Collection $model
 44:      * @param mixed $attributes
 45:      */
 46:     public static function dynamic($model, &$attributes = null)
 47:     {
 48:         // dynamic migration is not allowed when attributes are locked explicitly
 49:         $table_schema = Cache::get($model->getTable());
 50: 
 51:         if (isset($table_schema['lock_attributes']) && $table_schema['lock_attributes']) {
 52:             return true;
 53:         }
 54: 
 55:         // Cached attributes
 56:         $cached_attributes = array_map(function($item) { return $item['name']; }, isset($table_schema['attributes']) ? $table_schema['attributes'] : array());
 57: 
 58:         $config = array(
 59:             'dynamic' => true,
 60:             'attributes' => array()
 61:         );
 62: 
 63:         foreach ($attributes as $field => $value) {
 64:             // extract datatype from field value to migrate
 65:             $datatype = strtolower(gettype($value));
 66: 
 67:             // ignore 'null' data-types
 68:             if ($datatype == 'null') {
 69:                 unset($attributes[$field]);
 70:                 continue;
 71:             }
 72: 
 73:             // Detect large text blocks to declare 'text' datatype.
 74:             if ($datatype == 'string' && strlen($value) > 255) {
 75:                 $datatype = 'text';
 76:             }
 77: 
 78:             if ($datatype !== 'array' && !in_array($field, $cached_attributes)) {
 79:                 $config['attributes'][] = array('name' => $field, 'type' => $datatype);
 80:             }
 81:         }
 82: 
 83:         if (count($config['attributes']) > 0) {
 84:             return static::migrate($model, $config);
 85:         } else {
 86:             return false;
 87:         }
 88:     }

 91:      * migrate
 92:      *
 93:      * @param Hook\Model\Collection $model
 94:      * @param array $config
 95:      *
 96:      * @return bool
 97:      */
 98:     public static function migrate($model, $config)
 99:     {
100:         $result = false;
101:         $connection = $model->getConnectionResolver()->connection();
102: 
103:         // Ignore NoSQL databases.
104:         if (!$connection->getPdo()) { return; }
105: 
106:         // Get modified Schema\Grammar for hook features.
107:         $connection->setSchemaGrammar(static::getSchemaGrammar($connection));
108: 
109:         // Set custom blueprint resolver
110:         $builder = $connection->getSchemaBuilder();
111:         $builder->blueprintResolver(function($table, $callback) {
112:             return new \Hook\Database\Schema\Blueprint($table, $callback);
113:         });
114: 
115:         $table = $model->getTable();
116:         $table_prefix = Context::getPrefix();
117: 
118:         $is_creating = (!$builder->hasTable($table));
119:         $method = ($is_creating) ? 'create' : 'table';
120: 
121:         $table_schema = Cache::get($table);
122: 
123:         // sanitize / normalize relationship definitions
124:         if (isset($config['relationships'])) {
125:             foreach($config['relationships'] as $relation => $fields) {
126:                 $config['relationships'][$relation] = Relation::sanitize($relation, $fields);
127:             }
128:         } else {
129:             $config['relationships'] = array();
130:         }
131: 
132:         if (!isset($config['attributes'])) {
133:             $config['attributes'] = array();
134:         }
135: 
136:         if (!empty($config['attributes']) || !empty($config['relationships'])) {
137:             $migrate = function ($t) use (&$table, &$table_prefix, &$builder, &$is_creating, &$table_schema, $config, &$result) {
138:                 if ($is_creating) {
139:                     $t->increments('_id'); // primary key
140:                     $t->timestamps();      // created_at / updated_at field
141:                     $t->softDeletes();     // deleted_at field
142:                 } else {
143:                     $table_columns = $builder->getColumnListing($table);
144:                 }
145: 
146:                 foreach($config['attributes'] as $attribute) {
147:                     if (!isset($attribute['name'])) {
148:                         throw new MethodFailureException('invalid_schema');
149:                     }
150: 
151:                     $field_name = strtolower(array_remove($attribute, 'name'));
152:                     $type = camel_case(array_remove($attribute, 'type') ?: 'string');
153: 
154:                     $default = array_remove($attribute, 'default');
155:                     $index = array_remove($attribute, 'index');
156:                     $unique = array_remove($attribute, 'unique') || $index == 'unique';
157:                     $required = array_remove($attribute, 'required');
158: 
159:                     // spatial indexes are NOT NULL by default
160:                     $nullable = !$required && ($type !== 'point');
161: 
162:                     // Skip default fields
163:                     $ignore_fields = array('created_at', 'updated_at', 'deleted_at');
164:                     if (in_array($field_name, $ignore_fields)) {
165:                         continue;
166:                     }
167: 
168:                     // Skip if column already exists
169:                     if (!$is_creating && in_array($field_name, array_map('strtolower', $table_columns))) {
170:                         continue;
171:                     }
172: 
173:                     if (count($attribute) > 0) {
174:                         // the remaining attributes on field definition are
175:                         // the data-type related config, such as 'length',
176:                         // 'allowed', 'total', 'places', etc.
177:                         $column = $t->newColumn($type, $field_name, $attribute);
178:                     } else {
179:                         $column = $t->{$type}($field_name);
180:                     }
181: 
182:                     // apply default value
183:                     if ($default) {
184:                         $required = true;
185:                         $column->default($default);
186:                     }
187: 
188:                     // columns are nullable unless specified as 'required'
189:                     if ($nullable) { $column->nullable(); }
190: 
191:                     if ($index == 'spatial') {
192:                         // apply geospatial index, only MyISQL
193:                         $t->spatialIndex($field_name);
194:                     } else if ($index && !$unique) {
195:                         // apply index if specified
196:                         $column->index();
197:                     }
198: 
199:                     if ($unique) {
200:                         // apply unique index if specified
201:                         $unique_fields = (!is_array($unique)) ? $field_name : array_unique(array_merge(array($field_name), $unique));
202:                         $t->unique($unique_fields);
203:                     }
204:                 }
205: 
206:                 foreach($config['relationships'] as $relation => $fields) {
207:                     // only create field on belongs_to relationships
208:                     if ($relation == "belongs_to") {
209:                         foreach ($fields as $field => $collection) {
210:                             $foreign_field = $field . '_id';
211: 
212:                             // skip if field already exists
213:                             if ($builder->hasColumn($table, $foreign_field)) {
214:                                 continue;
215:                             }
216: 
217:                             // maybe 'collection' table isn't created here.
218:                             // TODO: create related table before referencing foreign key.
219:                             $column = $t->unsignedInteger($foreign_field)->index();
220:                             $column->nullable();
221: 
222:                             // $t->foreign($foreign_field)
223:                             //     ->references('_id')
224:                             //     ->on($table_prefix . $collection);
225:                         }
226:                     }
227:                 }
228: 
229:                 // return true when any modification is present
230:                 if (count($t->getColumns()) > 0 || count($t->getCommands()) > 0) {
231:                     $result = true;
232:                 }
233: 
234:             };
235: 
236:             call_user_func(array($builder, $method), $table_prefix . $table, $migrate);
237:         }
238: 
239:         // Cache table schema for further reference
240:         $table_schema = array_merge_recursive($table_schema, $config);
241:         Cache::forever($table, $table_schema);
242: 
243:         $app_collections = Cache::get('app_collections');
244:         Cache::forever('app_collections', array_unique(array_merge($app_collections, array($table))));
245: 
246:         return $result;
247:     }
