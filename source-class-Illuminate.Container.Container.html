411:         // just return an existing instance instead of instantiating new instances

482:     /**
483:      * Instantiate a concrete instance of the given type.
484:      *
485:      * @param  string  $concrete
486:      * @param  array   $parameters
487:      * @return mixed
488:      *
489:      * @throws BindingResolutionException
490:      */
491:     public function build($concrete, $parameters = array())
492:     {
493:         // If the concrete type is actually a Closure, we will just execute it and
494:         // hand back the results of the functions, which allows functions to be
495:         // used as resolvers for more fine-tuned resolution of these objects.
496:         if ($concrete instanceof Closure)
497:         {
498:             return $concrete($this, $parameters);
499:         }
500: 
501:         $reflector = new ReflectionClass($concrete);
502: 
503:         // If the type is not instantiable, the developer is attempting to resolve
504:         // an abstract type such as an Interface of Abstract Class and there is
505:         // no binding registered for the abstractions so we need to bail out.
506:         if ( ! $reflector->isInstantiable())
507:         {
508:             $message = "Target [$concrete] is not instantiable.";
509: 
510:             throw new BindingResolutionException($message);
511:         }
512: 
513:         $constructor = $reflector->getConstructor();
514: 
515:         // If there are no constructors, that means there are no dependencies then
516:         // we can just resolve the instances of the objects right away, without
517:         // resolving any other types or dependencies out of these containers.
518:         if (is_null($constructor))
519:         {
520:             return new $concrete;
521:         }
522: 
523:         $dependencies = $constructor->getParameters();
524: 
525:         // Once we have all the constructor's parameters we can create each of the
526:         // dependency instances and then use the reflection instances to make a
527:         // new instance of this class, injecting the created dependencies in.
528:         $parameters = $this->keyParametersByArgument(
529:             $dependencies, $parameters
530:         );
531: 
532:         $instances = $this->getDependencies(
533:             $dependencies, $parameters
534:         );
535: 
536:         return $reflector->newInstanceArgs($instances);
537:     }

539:     /**
540:      * Resolve all of the dependencies from the ReflectionParameters.
541:      *
542:      * @param  array  $parameters
543:      * @param  array  $primitives
544:      * @return array
545:      */
546:     protected function getDependencies($parameters, array $primitives = array())
547:     {
548:         $dependencies = array();
549: 
550:         foreach ($parameters as $parameter)
551:         {
552:             $dependency = $parameter->getClass();
553: 
554:             // If the class is null, it means the dependency is a string or some other
555:             // primitive type which we can not resolve since it is not a class and
556:             // we will just bomb out with an error since we have no-where to go.
557:             if (array_key_exists($parameter->name, $primitives))
558:             {
559:                 $dependencies[] = $primitives[$parameter->name];
560:             }
561:             elseif (is_null($dependency))
562:             {
563:                 $dependencies[] = $this->resolveNonClass($parameter);
564:             }
565:             else
566:             {
567:                 $dependencies[] = $this->resolveClass($parameter);
568:             }
569:         }
570: 
571:         return (array) $dependencies;
572:     }
