 339:      * Add a basic where clause to the query.
 340:      *
 341:      * @param  string  $column
 342:      * @param  string  $operator
 343:      * @param  mixed   $value
 344:      * @param  string  $boolean
 345:      * @return \Illuminate\Database\Query\Builder|static
 346:      *
 347:      * @throws \InvalidArgumentException
 348:      */
 349:     public function where($column, $operator = null, $value = null, $boolean = 'and')
 350:     {
 351:         if (func_num_args() == 2)
 352:         {
 353:             list($value, $operator) = array($operator, '=');
 354:         }
 355:         elseif ($this->invalidOperatorAndValue($operator, $value))
 356:         {
 357:             throw new \InvalidArgumentException("Value must be provided.");
 358:         }
 359: 
 360:         // If the columns is actually a Closure instance, we will assume the developer
 361:         // wants to begin a nested where statement which is wrapped in parenthesis.
 362:         // We'll add that Closure to the query then return back out immediately.
 363:         if ($column instanceof Closure)
 364:         {
 365:             return $this->whereNested($column, $boolean);
 366:         }
 367: 
 368:         // If the given operator is not found in the list of valid operators we will
 369:         // assume that the developer is just short-cutting the '=' operators and
 370:         // we will set the operators to '=' and set the values appropriately.
 371:         if ( ! in_array(strtolower($operator), $this->operators, true))
 372:         {
 373:             list($value, $operator) = array($operator, '=');
 374:         }
 375: 
 376:         // If the value is a Closure, it means the developer is performing an entire
 377:         // sub-select within the query and we will need to compile the sub-select
 378:         // within the where clause to get the appropriate query record results.
 379:         if ($value instanceof Closure)
 380:         {
 381:             return $this->whereSub($column, $operator, $value, $boolean);
 382:         }
 383: 
 384:         // If the value is "null", we will just assume the developer wants to add a
 385:         // where null clause to the query. So, we will allow a short-cut here to
 386:         // that method for convenience so the developer doesn't have to check.
 387:         if (is_null($value))
 388:         {
 389:             return $this->whereNull($column, $boolean, $operator != '=');
 390:         }
 391: 
 392:         // Now that we are working with just a simple query we can put the elements
 393:         // in our array and add the query binding to our array of bindings that
 394:         // will be bound to each SQL statements when it is finally executed.
 395:         $type = 'Basic';
 396: 
 397:         $this->wheres[] = compact('type', 'column', 'operator', 'value', 'boolean');
 398: 
 399:         if ( ! $value instanceof Expression)
 400:         {
 401:             $this->bindings[] = $value;
 402:         }
 403: 
 404:         return $this;
 405:     }
