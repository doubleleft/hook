306:     /**
307:      * Copy a directory from one location to another.
308:      *
309:      * @param  string  $directory
310:      * @param  string  $destination
311:      * @param  int     $options
312:      * @return bool
313:      */
314:     public function copyDirectory($directory, $destination, $options = null)
315:     {
316:         if ( ! $this->isDirectory($directory)) return false;
317: 
318:         $options = $options ?: FilesystemIterator::SKIP_DOTS;
319: 
320:         // If the destination directory does not actually exist, we will go ahead and
321:         // create it recursively, which just gets the destination prepared to copy
322:         // the files over. Once we make the directory we'll proceed the copying.
323:         if ( ! $this->isDirectory($destination))
324:         {
325:             $this->makeDirectory($destination, 0777, true);
326:         }
327: 
328:         $items = new FilesystemIterator($directory, $options);
329: 
330:         foreach ($items as $item)
331:         {
332:             // As we spin through items, we will check to see if the current file is actually
333:             // a directory or a file. When it is actually a directory we will need to call
334:             // back into this function recursively to keep copying these nested folders.
335:             $target = $destination.'/'.$item->getBasename();
336: 
337:             if ($item->isDir())
338:             {
339:                 $path = $item->getPathname();
340: 
341:                 if ( ! $this->copyDirectory($path, $target, $options)) return false;
342:             }
343: 
344:             // If the current items is just a regular file, we will just copy this to the new
345:             // location and keep looping. If for some reason the copy fails we'll bail out
346:             // and return false, so the developer is aware that the copy process failed.
347:             else
348:             {
349:                 if ( ! $this->copy($item->getPathname(), $target)) return false;
350:             }
351:         }
352: 
353:         return true;
354:     }

366:     {
367:         if ( ! $this->isDirectory($directory)) return false;
368: 
369:         $items = new FilesystemIterator($directory);
370: 
371:         foreach ($items as $item)
372:         {
373:             // If the item is a directory, we can just recurse into the function and
374:             // delete that sub-director, otherwise we'll just delete the file and
375:             // keep iterating through each file until the directory is cleaned.
376:             if ($item->isDir())
377:             {
378:                 $this->deleteDirectory($item->getPathname());
379:             }
380: 
381:             // If the item is just a file, we can go ahead and delete it since we're
382:             // just looping through and waxing all of the files in this directory
383:             // and calling directories recursively, so we delete the real path.
384:             else
385:             {
386:                 $this->delete($item->getPathname());
387:             }
388:         }
389: 
390:         if ( ! $preserve) @rmdir($directory);
391: 
392:         return true;
393:     }
