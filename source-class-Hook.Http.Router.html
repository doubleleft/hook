 96:     public static function mount($path, $controller_klass)
 97:     {
 98:         $methods = get_class_methods($controller_klass);
 99: 
100:         // skip
101:         if (!$methods) {
102:             debug("'{$controller_klass}' has no methods.");
103:             return;
104:         }
105: 
106:         foreach($methods as $method_name) {
107:             // skip invalid methods
108:             if ($method_name == '__construct') {
109:                 continue;
110:             }
111: 
112:             // call 'mounted' method
113:             if ($method_name == 'mounted') {
114:                 call_user_func(array($controller_klass, 'mounted'), $path);
115:                 continue;
116:             }
117: 
118:             preg_match_all('/^(get|put|post|patch)(.*)/', $method_name, $matches);
119:             $has_matches = (count($matches[1]) > 0);
120: 
121:             $http_method = $has_matches ? $matches[1][0] : 'any';
122:             $route_name = $has_matches ? $matches[2][0] : $method_name;
123: 
124:             $route = str_finish($path, '/');
125:             if ($route_name !== 'index') {
126:                 $route .= snake_case($route_name);
127:             }
128: 
129:             static::$instance->{$http_method}($route, "{$controller_klass}:{$method_name}");
