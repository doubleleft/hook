105:     protected function makeConnection($name)
106:     {
107:         $config = $this->getConfig($name);
108: 
109:         // First we will check by the connection name to see if an extension has been
110:         // registered specifically for that connection. If it has we will call the
111:         // Closure and pass it the config allowing it to resolve the connection.
112:         if (isset($this->extensions[$name]))
113:         {
114:             return call_user_func($this->extensions[$name], $config, $name);
115:         }
116: 
117:         $driver = $config['driver'];
118: 
119:         // Next we will check to see if an extension has been registered for a driver
120:         // and will call the Closure if so, which allows us to have a more generic
121:         // resolver for the drivers themselves which applies to all connections.
122:         if (isset($this->extensions[$driver]))
123:         {
124:             return call_user_func($this->extensions[$driver], $config, $name);
125:         }
126: 
127:         return $this->factory->make($config, $name);
128:     }
