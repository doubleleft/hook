137:     /**
138:      * Get the current page for the request.
139:      *
140:      * @param  int $lastPage
141:      * @return int
142:      */
143:     protected function calculateCurrentPage($lastPage)
144:     {
145:         $page = $this->env->getCurrentPage();
146: 
147:         // The page number will get validated and adjusted if it either less than one
148:         // or greater than the last page available based on the count of the given
149:         // items array. If it's greater than the last, we'll give back the last.
150:         if (is_numeric($page) && $page > $lastPage) {
151:             return $lastPage > 0 ? $lastPage : 1;
152:         }
153: 
154:         return $this->isValidPageNumber($page) ? (int) $page : 1;
155:     }
